### 2. Реализация ETL-пайплайна

Класс `SalesETLPipeline`:

```python
import pandas as pd
import numpy as np
import sqlite3
import matplotlib.pyplot as plt
import seaborn as sns
from datetime import datetime
from sqlalchemy import create_engine
import logging
from sqlalchemy import create_engine, text

# Настройка логирования
logging.basicConfig(level=logging.INFO, format='%(asctime)s - %(levelname)s - %(message)s')
logger = logging.getLogger(__name__)


class SalesETLPipeline:
    """ETL пайплайн для обработки данных о продажах"""
    
    def __init__(self, csv_path, db_path='sales.db'):
        self.csv_path = csv_path
        self.db_path = db_path
        self.raw_data = None
        self.cleaned_data = None
        self.aggregated_data = None
        
    def extract(self):

        # Этап 1: Извлечение данных из CSV-файла

        logger.info("Начало этапа EXTRACT")
        
        try:
            self.raw_data = pd.read_csv(self.csv_path)
            logger.info(f"Загружено {len(self.raw_data)} строк, {len(self.raw_data.columns)} колонок")
        except FileNotFoundError:
            logger.error(f"Файл {self.csv_path} не найден")
            raise
        except pd.errors.EmptyDataError:
            logger.error("Файл пуст")
            raise
        except Exception as e:
            logger.error(f"Ошибка загрузки")
        return self.raw_data
    
    def transform(self):

        # Этап 2: Трансформация и очистка данных

        logger.info("Начало этапа TRANSFORM")
        
        df_raw = self.raw_data.copy()
        
        df = df_raw.drop_duplicates()
        
        numeric_cols = ['quantity', 'price_per_unit']
        for col in numeric_cols:
            median_val = df[col].median()
            df[col] = df[col].fillna(median_val)
            
        text_cols = ['category', 'product_name', 'customer_name']
        for col in text_cols:
            df[col] = df[col].fillna('Unknown')
        
        logger.info(f"Заполнено пропусков: числовые - {df[numeric_cols].isna().sum().sum()}")
        
        initial_len = len(df)
        df = df[(df['quantity'] > 0) & (df['price_per_unit'] > 0)]
        removed = initial_len - len(df)
        logger.info(f"Удалено аномальных строк: {removed}")
                
        df['order_date'] = pd.to_datetime(df['order_date'], errors='coerce')
        df['quantity'] = pd.to_numeric(df['quantity'], errors='coerce')
        df['price_per_unit'] = pd.to_numeric(df['price_per_unit'], errors='coerce')
        
        df['total_amount'] = df['quantity'] * df['price_per_unit']
        
        df['month_year'] = df['order_date'].dt.strftime('%Y-%m')
        
        self.cleaned_data = df
        logger.info(f"После очистки: {len(df)} строк")
        
        return self.cleaned_data
    
    def aggregate(self):
        
        # Этап 3: Агрегация данных для аналитики

        logger.info("Начало этапа AGGREGATE")
        
        df = self.cleaned_data.copy()
        
        self.aggregated_data = df.groupby(['category', 'month_year']).agg({
            'quantity': 'sum',
            'total_amount': 'sum',
            'price_per_unit': 'mean',
            'order_id': 'nunique'
        }).rename(columns={
            'quantity': 'total_quantity',
            'total_amount': 'total_revenue',
            'price_per_unit': 'avg_price',
            'order_id': 'order_count'
        }).reset_index()
        
        return self.aggregated_data
    
    def load_to_sqlite(self):

        # Этап 4: Загрузка данных в SQLite базу данных

        logger.info("Начало этапа LOAD")
        
        # Создание подключения
        engine = create_engine(f'sqlite:///{self.db_path}')
        
        self.cleaned_data.to_sql('sales_cleaned', engine, if_exists='replace', index=False)
        self.aggregated_data.to_sql('sales_aggregated', engine, if_exists='replace', index=False)
        with engine.connect() as conn:
            result = conn.execute(text("SELECT name FROM sqlite_master WHERE type='table';"))
            tables = result.fetchall()
            logger.info(f"Таблицы в БД: {[t[0] for t in tables]}")
        logger.info(f"Данные загружены в {self.db_path}")
        
    def visualize(self):
        logger.info("Начало этапа VISUALIZE")
        
        # 1. График выручки по категориям (суммарно за весь период)
        category_revenue = self.aggregated_data.groupby('category')['total_revenue'].sum().sort_values(ascending=False)
        
        plt.figure(figsize=(10, 6))
        category_revenue.plot(kind='bar', color='skyblue')
        plt.title('Выручка по категориям товаров', fontsize=14)
        plt.xlabel('Категория')
        plt.ylabel('Выручка (руб.)')
        plt.xticks(rotation=45)
        plt.tight_layout()
        plt.show()
        
        # 2. Динамика продаж по месяцам (линейный график с разбивкой по категориям)
        pivot_revenue = self.aggregated_data.pivot(index='month_year', columns='category', values='total_revenue')
        
        plt.figure(figsize=(12, 6))
        for column in pivot_revenue.columns:
            plt.plot(pivot_revenue.index, pivot_revenue[column], marker='o', label=column)
        plt.title('Динамика выручки по месяцам и категориям', fontsize=14)
        plt.xlabel('Месяц')
        plt.ylabel('Выручка (руб.)')
        plt.legend(title='Категория')
        plt.grid(True, linestyle='--', alpha=0.7)
        plt.xticks(rotation=45)
        plt.tight_layout()
        plt.show()
        
        # 3. Круговая диаграмма — доля каждой категории в общей выручке
        plt.figure(figsize=(8, 8))
        plt.pie(category_revenue, labels=category_revenue.index, autopct='%1.1f%%', startangle=90, 
                colors=['#ff9999','#66b3ff','#99ff99','#ffcc99'])
        plt.title('Доля категорий в общей выручке', fontsize=14)
        plt.axis('equal')
        plt.tight_layout()
        plt.show()
        
    def run(self):
        """Запуск полного ETL-пайплайна"""
        logger.info("=" * 50)
        logger.info("ЗАПУСК ETL ПАЙПЛАЙНА")
        logger.info("=" * 50)
        
        self.extract()
        self.transform()
        self.aggregate()
        self.load_to_sqlite()
        self.visualize()
        
        logger.info("ETL пайплайн успешно завершён")


if __name__ == "__main__":
    pipeline = SalesETLPipeline('data/sales.csv', 'sales.db')
    pipeline.run()
```

### 4. Ответы на вопросы

**1. Какие аномалии и пропуски были обнаружены в исходных данных? Как вы их обработали и почему?**  

`В исходных данных были обнаружены дубликация данных, проблемные и невозможные данные. Данные были отфильтрованы по уникальности и возможности значений, для того чтобы передавать в БД заранее подготовленные данные.`

**2. В чём разница между ETL и ELT? Какой подход для данной задачи?**  

`Разница ETL и ELT заключается в последовательности действий. ETL - классический метод, работающий по принципу загрузки данных, их предварительной очистки и стандартизации, после чего внесения в базу данных. ELT - новый метод, который после загрузки данных сразу направляет их в БД, где данные и стандартизируются. В данной задаче был использован подход ETL - загрузка данных из CSV > трансформация через python pipeline > добавление данных в БД.`

**3. Почему в реальных проектах данные не загружают напрямую в базу без очистки?**  

`Опасно полагаться на пользователя или другого продюссера как на источник стандартизированной информации, без контроля над потоком данных в виде очистки и стандартизации данных, система может себя вести непредсказуемо, что в лучшем случае приведёт к багам, а в худшем - падению системы и безопасности.`

### 2. Продюсер событий (файл `producer.py`)


```python
from kafka import KafkaProducer
import json
import random
import time
import uuid
from datetime import datetime
import logging

logging.basicConfig(level=logging.INFO, format='%(asctime)s - %(levelname)s - %(message)s')
logger = logging.getLogger(__name__)


class OrderEventProducer:
    """Продюсер событий заказов в Kafka"""
    
    def __init__(self, bootstrap_servers='localhost:9092', topic='orders'):
        self.bootstrap_servers = bootstrap_servers
        self.topic = topic
        self.producer = None
        
    def connect(self):
        """Создание подключения к Kafka"""
        try:
            self.producer = KafkaProducer(
                bootstrap_servers=self.bootstrap_servers,
                value_serializer=lambda v: json.dumps(v).encode('utf-8'),
                key_serializer=lambda k: k.encode('utf-8') if k else None
            )
            logger.info(f"Connected to Kafka at {self.bootstrap_servers}")
        except Exception as e:
            logger.error(f"Failed to connect: {e}")
            raise
    
    def generate_order(self):
        """Генерация случайного заказа"""
        products = [
            {"product_id": 1, "name": "Ноутбук", "price": 75000, "category": "Электроника"},
            {"product_id": 2, "name": "Мышь", "price": 1500, "category": "Электроника"},
            {"product_id": 3, "name": "Книга SQL", "price": 2500, "category": "Книги"},
            {"product_id": 4, "name": "Клавиатура", "price": 5000, "category": "Электроника"},
            {"product_id": 5, "name": "Монитор", "price": 25000, "category": "Электроника"},
            {"product_id": 6, "name": "Книга Python", "price": 3500, "category": "Книги"}
        ]
        
        customers = [
            {"id": 1, "name": "Анна Смирнова", "city": "Москва"},
            {"id": 2, "name": "Петр Иванов", "city": "СПб"},
            {"id": 3, "name": "Мария Сидорова", "city": "Казань"},
            {"id": 4, "name": "Иван Петров", "city": "Москва"},
            {"id": 5, "name": "Елена Козлова", "city": "Новосибирск"}
        ]
        
        product = random.choice(products)
        customer = random.choice(customers)
        quantity = random.randint(1, 3)
        
        order = {
            "order_id": str(uuid.uuid4())[:8],
            "timestamp": datetime.now().isoformat(),
            "customer": customer,
            "items": [{
                "product_id": product["product_id"],
                "product_name": product["name"],
                "category": product["category"],
                "quantity": quantity,
                "unit_price": product["price"],
                "total_price": quantity * product["price"]
            }],
            "total_amount": quantity * product["price"],
            "payment_method": random.choice(["card", "cash", "online"])
        }
        
        return order
    
    def send_order(self, order):
        """Отправка одного заказа в Kafka"""
        key = str(order['customer']['id'])
        future = self.producer.send(self.topic, key=key, value=order)
        
        # Ожидание подтверждения
        try:
            record_metadata = future.get(timeout=10)
            logger.info(f"Order {order['order_id']} sent to partition {record_metadata.partition} at offset {record_metadata.offset}")
        except Exception as e:
            logger.error(f"Failed to send order: {e}")
        
        return future
    
    def run(self, interval_seconds=2, max_orders=10):
        """Запуск генерации и отправки заказов"""
        logger.info(f"Starting producer. Sending {max_orders} orders every {interval_seconds}s")
        
        self.connect()
        
        for i in range(max_orders):
            order = self.generate_order()
            self.send_order(order)
            time.sleep(interval_seconds)
        
        # Ожидание завершения всех отправок
        self.producer.flush()
        logger.info("All orders sent successfully")
        self.producer.close()
        
    def connect(self):
        self.producer = KafkaProducer(
            bootstrap_servers=self.bootstrap_servers,
            value_serializer=lambda v: json.dumps(v).encode('utf-8'),
            key_serializer=lambda k: k.encode('utf-8') if k else None,
            acks='all',
            retries=3
        )


if __name__ == "__main__":
    producer = OrderEventProducer()
    producer.run(interval_seconds=1, max_orders=15)
```

            self.stats['recent_orders'].pop(0)
    
    def print_stats(self):
        """Вывод текущей статистики в консоль"""
        logger.info("=" * 50)
        logger.info(f"ТЕКУЩАЯ СТАТИСТИКА ЗАКАЗОВ")
        logger.info(f"Всего заказов: {self.stats['total_orders']}")
        logger.info(f"Общая выручка: {self.stats['total_revenue']:,.2f} руб.")
        logger.info(f"Средний чек: {self.stats['total_revenue'] / self.stats['total_orders']:,.2f} руб." if self.stats['total_orders'] > 0 else "Средний чек: 0")
        logger.info(f"Заказов по категориям: {dict(self.stats['orders_by_category'])}")
        logger.info(f"Заказов по городам: {dict(self.stats['orders_by_city'])}")
        logger.info(f"Последние 3 заказа: {self.stats['recent_orders'][-3:]}")
        logger.info("=" * 50)
    
    def run(self, timeout_ms=1000):
        """Запуск консюмера для непрерывного чтения сообщений"""
        logger.info("Starting consumer. Waiting for orders...")
        self.connect()
        
        try:
            for message in self.consumer:
                order = message.value
                logger.info(f"Received order {order['order_id']} from customer {order['customer']['name']}")
                
                self.update_stats(order)
                self.print_stats()
                
        except KeyboardInterrupt:
            logger.info("Consumer stopped by user")
        finally:
            self.consumer.close()
            self.print_final_report()
    
    def print_final_report(self):
        """Финальный отчёт после остановки консюмера"""
        runtime = datetime.now() - self.stats['start_time']

from datetime import datetime
from collections import defaultdict

class WindowedStatsConsumer(OrderStatsConsumer):
    """Консюмер со скользящим окном для расчёта статистики за последнюю минуту"""
    
    def __init__(self, *args, window_seconds=60, **kwargs):
        super().__init__(*args, **kwargs)
        self.window_seconds = window_seconds
        self.order_records = []  #(timestamp, order_amount, categories_list, city, order_info)
    
    def update_stats(self, order):
        """Добавляет заказ в окно и пересчитывает статистику"""
        current_timestamp = datetime.now().timestamp()
        
        amount = order['total_amount']
        categories = [item['category'] for item in order['items']]
        city = order['customer']['city']
        order_info = {
            'order_id': order['order_id'],
            'customer': order['customer']['name'],
            'total': amount,
            'time': order['timestamp']
        }

        self.order_records.append((current_timestamp, amount, categories, city, order_info))

        self.cleanup_old_orders()

        self.recalculate_stats()
    
    def cleanup_old_orders(self):
        """Удаляет заказы, время жизни которых превысило window_seconds"""
        current_time = datetime.now().timestamp()
        cutoff = current_time - self.window_seconds
        self.order_records = [rec for rec in self.order_records if rec[0] >= cutoff]
    
    def recalculate_stats(self):
        """Пересчитывает агрегаты из текущих записей в окне"""
        self.stats['total_orders'] = len(self.order_records)
        self.stats['total_revenue'] = sum(rec[1] for rec in self.order_records)
        self.stats['orders_by_category'].clear()
        self.stats['orders_by_city'].clear()
        self.stats['recent_orders'] = []
        
        for rec in self.order_records:
            _, _, categories, city, order_info = rec
            for cat in categories:
                self.stats['orders_by_category'][cat] += 1
            self.stats['orders_by_city'][city] += 1
            self.stats['recent_orders'].append(order_info)
        
        if len(self.stats['recent_orders']) > 10:
            self.stats['recent_orders'] = self.stats['recent_orders'][-10:]

if __name__ == "__main__":
    consumer = OrderStatsConsumer()
    consumer.run()

### 5. Ответы на вопросы

**1. Что такое топик, партиция, оффсет в Kafka? Как они использовались в работе?**  

`Топик - наименование канала передачи информации между продюссером и консьюмером. Партиция - область топика для хранения информации в которые попадат данные относительно ключей. Оффсет - идентификационный номер для определения места данных в партиции.`

**2. В чём разница между продюсером и консюмером? Зачем нужна группа консюмеров (`group_id`)?**  

`Продюссер генерирует запросы, консьюмер получает запросы. Группа консьюмеров нужна для стабильности системы обработки информации и максимальной производительности.`

**3. Какие преимущества даёт потоковая обработка перед пакетной (ETL из Части 1)?**  

`Главное преимущество потоковой обработки перед пакетной это минимальная задержка. В пакетном ETL статистика обновляется только после запуска скрипта, тогда как консюмер в Kafka обновляет сумму заказов мгновенно после поступления каждого сообщения.`

**4. Приведите 3 реальных сценария использования Kafka в бизнесе (не связанных с логированием).**  

`Онлайн-рекомендации в интернет-магазине: пользователь просматривает товары > клики отправляются в Kafka в реальном времени > потоковый консюмер анализирует поведение за последние 5 минут и обновляет персональные рекомендации на сайте без перезагрузки страницы.`

`Обнаружение мошеннических транзакций: банк отправляет все операции по картам в Kafka > консюмеры проверяют каждую транзакцию за миллисекунды > страна отличается от предыдущей → моментальный блокировочный сигнал и SMS клиенту.`

`Система управления складом и инвентаризацией: сканер штрих-кода на складе при отгрузке товара отправляет событие в Kafka > несколько консюмеров параллельно обновляют > избегаем двойных продаж и держим остатки актуальными в реальном времени.`

**5. Что произойдёт, если запустить 2 консюмера с одинаковым `group_id`?**  

`Если запустить два экземпляра консюмера с одинаковым group_id и подписанных на один и тот же топик, Kafka автоматически разделит партиции топика между этими консюмерами.`

## Сравнение ETL и потоковой обработки (по итогам лабораторной)

| Характеристика | ETL (Python + SQLite) | Потоковая обработка (Kafka) |
|----------------|------------------------|------------------------------|
| Тип обработки | Пакетная (batch) | Потоковая (streaming) |
| Задержка | Минуты/часы | Миллисекунды/секунды |
| Хранение состояний | На диске (SQLite) | В памяти консюмера |
| Масштабирование | Вертикальное | Горизонтальное (партиции) |
| Отказоустойчивость | Низкая | Высокая (репликация) |
| Сложность реализации | Низкая | Средняя |
| Use cases | Ежедневные отчёты, дашборды | Мониторинг, алерты, real-time analytics |
